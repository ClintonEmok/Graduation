    Add hasOverlappingIntervals function using sort + sweep algorithm:
    ```typescript
    function hasOverlappingIntervals(intervals: { startPercent: number; endPercent: number }[]): boolean {
      if (intervals.length <= 1) return false;
      const sorted = [...intervals].sort((a, b) => a.startPercent - b.startPercent);
      for (let i = 1; i < sorted.length; i++) {
        if (sorted[i].startPercent < sorted[i - 1].endPercent) {
          return true; // Overlap detected - use strict < not <=
        }
      }
      return false;
    }
    ```
