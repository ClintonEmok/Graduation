Create `src/workers/kdeSlice.worker.ts` following the pattern established by `src/workers/adaptiveTime.worker.ts`:

**Worker contract:**

```typescript
// Input — sent via postMessage from main thread
export interface KdeWorkerInput {
  requestId: number;
  sliceIndex: number;
  points: Array<{ x: number; z: number }>;
  params?: Partial<KdeParams>;
}

// Output — sent back via postMessage with transferable buffer
export interface KdeWorkerOutput {
  requestId: number;
  sliceIndex: number;
  cellsBuffer: ArrayBuffer;  // transferable Float32Array.buffer
  cellCount: number;
  maxIntensity: number;
}
```

**Worker implementation:**

1. Import `computeSliceKde` from `@/lib/kde` and `KdeParams` from `@/lib/kde/types`
2. Set up `self.onmessage` handler:
   - Receive `KdeWorkerInput`
   - Call `computeSliceKde(points, params)`
   - Serialize result cells to a flat `Float32Array` (4 floats per cell: x, z, intensity, support)
   - Transfer the `.buffer` via postMessage's transfer list
   - Include `cellCount` and `maxIntensity` in output for deserialization

**Key implementation details:**

```typescript
// Serialize KDE cells to transferable flat buffer
const cellCount = result.cells.length;
const flat = new Float32Array(cellCount * 4);
for (let i = 0; i < cellCount; i++) {
  flat[i * 4] = result.cells[i].x;
  flat[i * 4 + 1] = result.cells[i].z;
  flat[i * 4 + 2] = result.cells[i].intensity;
  flat[i * 4 + 3] = result.cells[i].support;
}

// Transfer ownership — zero copy
self.postMessage(
  {
    requestId,
    sliceIndex,
    cellsBuffer: flat.buffer,
    cellCount: cellCount,
    maxIntensity: result.maxIntensity,
  } satisfies KdeWorkerOutput,
  [flat.buffer] // transferable — buffer ownership moves to main thread
);
```

**Important:** The buffer transfer means the worker can't use `flat` after `postMessage`. This is intentional — the main thread takes ownership for zero-copy.

**No worker pool needed.** The sequential nature of slice KDE (ordered slices processed in batch) means a single worker instance is optimal. The worker is created lazily on first use and terminated on unmount.

**Step 2: Replace the synchronous useMemo (lines 161-188)**

Remove the entire `const { sliceKdes, countedSlices } = useMemo(...)` block (lines 161-188) and replace with:

```typescript
// Worker-based KDE computation
const workerRef = useRef<Worker | null>(null);
const [sliceKdes, setSliceKdes] = useState<KdeCell[][]>([]);
const [kdeComputing, setKdeComputing] = useState(false);

useEffect(() => {
  if (crimesBySlice.length === 0) {
    setSliceKdes([]);
    setKdeComputing(false);
    return;
  }

  // Lazily create worker
  if (!workerRef.current) {
    workerRef.current = new Worker(
      new URL('@/workers/kdeSlice.worker.ts', import.meta.url)
    );
  }

  setKdeComputing(true);
  let completed = 0;
  const results: (KdeCell[] | undefined)[] = new Array(crimesBySlice.length);

  workerRef.current.onmessage = (e: MessageEvent<KdeWorkerOutput>) => {
    const { sliceIndex, cellsBuffer, cellCount, maxIntensity } = e.data;
    const flat = new Float32Array(cellsBuffer);

    // Deserialize cells from flat array
    const cells: KdeCell[] = [];
    for (let i = 0; i < cellCount; i++) {
      cells.push({
        x: flat[i * 4],
        z: flat[i * 4 + 1],
        intensity: flat[i * 4 + 2],
        support: Math.round(flat[i * 4 + 3]),
      });
    }

    results[sliceIndex] = cells;
    completed += 1;

    if (completed === crimesBySlice.length) {
      setSliceKdes(results.map((r) => r ?? []));
      setKdeComputing(false);
    }
  };

  // Submit all slices in batch
  crimesBySlice.forEach((sliceCrimes, i) => {
    if (sliceCrimes.length === 0) {
      // Empty slice — complete immediately with empty array
      results[i] = [];
      completed += 1;
      if (completed === crimesBySlice.length) {
        setSliceKdes(results.map((r) => r ?? []));
        setKdeComputing(false);
      }
    } else {
      workerRef.current!.postMessage({
        requestId: i,
        sliceIndex: i,
        points: sliceCrimes.map((c) => ({ x: c.x, z: c.z })),
      });
    }
  });

  return () => {
    workerRef.current?.terminate();
    workerRef.current = null;
  };
}, [crimesBySlice]);
```
