```typescript
import type { KdeCell } from '@/lib/kde';

/** Linear interpolation between two values */
export function lerp(a: number, b: number, t: number): number {
  return a + (b - a) * t;
}

/** Easing functions for different transition types */
export const Easing = {
  /** Analytical state changes — quick start, slow finish */
  easeOutCubic(t: number): number {
    return 1 - Math.pow(1 - t, 3);
  },
  /** Camera/orientation transitions — smooth throughout */
  easeInOutCubic(t: number): number {
    return t < 0.5 ? 4 * t * t * t : 1 - Math.pow(-2 * t + 2, 3) / 2;
  },
  /** Quick transitions for active slice emphasis */
  power2InOut(t: number): number {
    return t < 0.5 ? 2 * t * t : 1 - Math.pow(-2 * t + 2, 2) / 2;
  },
};

/**
 * Interpolate between two arrays of KdeCell using CPU lerp.
 * Used for opt-in smooth interpolation between consecutive slices.
 * Map-based merge: cells with same (x,z) coordinate are lerped.
 */
export function interpolateKdeCells(
  from: KdeCell[],
  to: KdeCell[],
  t: number,
  threshold = 0.005
): KdeCell[] {
  // Build lookup for 'to' cells
  const toMap = new Map<string, KdeCell>();
  for (const cell of to) {
    toMap.set(`${cell.x}_${cell.z}`, cell);
  }

  return from
    .map((fromCell): KdeCell => {
      const key = `${fromCell.x}_${fromCell.z}`;
      const toCell = toMap.get(key);
      const intensity = toCell
        ? lerp(fromCell.intensity, toCell.intensity, t)
        : fromCell.intensity * (1 - t);
      return { ...fromCell, intensity: Number(intensity.toFixed(4)) };
    })
    .filter((cell) => cell.intensity > threshold);
}
```

**File 2: `src/lib/motion/aging.ts`**

Aging/decay utilities for per-slice opacity trails:

```typescript
/** Configuration for aging/opacity decay */
export interface AgingConfig {
  /** Opacity drop-off per step away from active slice */
  decayPerStep: number; // default: 0.35
  /** Minimum opacity for visible but distant slices */
  floorOpacity: number; // default: 0.05
  /** Adjacent slices that get special "near" opacity */
  adjacentRadius: number; // default: 1
}

export const DEFAULT_AGING_CONFIG: AgingConfig = {
  decayPerStep: 0.35,
  floorOpacity: 0.05,
  adjacentRadius: 1,
};

/**
 * Calculate a slice's opacity multiplier based on distance from active.
 *
 * Opacity profile:
 *   distance 0 (active)     → 1.0
 *   distance 1 (adjacent)   → decayPerStep (0.35)
 *   distance 2              → 0.10
 *   distance 3+             → exponential decay to floorOpacity
 */
export function agingOpacity(
  sliceIndex: number,
  activeIndex: number,
  config: AgingConfig = DEFAULT_AGING_CONFIG
): number {
  const distance = Math.abs(sliceIndex - activeIndex);

  if (distance === 0) return 1;
  if (distance <= config.adjacentRadius) return config.decayPerStep;
  if (distance === 2) return 0.1;

  // Exponential decay for distant slices
  const decayed = config.decayPerStep * Math.pow(0.3, distance - 1);
  return Math.max(config.floorOpacity, decayed);
}

/**
 * Calculate trail intensity for accumulated frame blending.
 * Used in Phase 79 for temporal accumulation rendering.
 */
export function trailIntensity(
  frameAge: number,       // how many frames ago
  maxTrailLength = 8,
  decayRate = 0.15
): number {
  if (frameAge >= maxTrailLength) return 0;
  return Math.exp(-decayRate * frameAge);
}
```

Both modules are pure functions — no side effects, no React hooks, no Three.js imports. They can be unit tested independently.

```typescript
useEffect(() => {
  const controls = controlsRef.current;
  if (!controls || slices.length === 0) return;

  const targetY = yForIndex(activeIndex);
  const camPos = controls.camera.position.clone();

  gsap.to(camPos, {

    duration: 0.6,
    ease: 'power2.inOut',
    onUpdate: () => {
      controls.setLookAt(
        camPos.x,
        camPos.y,
        camPos.z,
        0,          // target.x — center of scene
        targetY,    // target.y — active slice center
        0,          // target.z — center of scene
        false,      // do not enable transition (GSAP handles it)
      );
    },
  });
}, [activeIndex, slices.length]);
```

**Important:** Do NOT call `gsap.to()` directly on `controlsRef.current` or any CameraControls method. CameraControls manages its own internal animation loop. Instead:
1. Animate a proxy `Vector3` (`camPos`)
2. Call `controls.setLookAt()` with the animated values in `onUpdate`
3. Pass `false` for the `enableTransition` parameter

This prevents GSAP's property animation from conflicting with CameraControls' internal state.
