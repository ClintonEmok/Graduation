    1. **Create `src/app/api/adaptive/contextual-baseline/route.ts`** — fallback server route.

       Structure (mirror `src/app/api/adaptive/global/route.ts:1-53` but with the contextual baseline query):
       ```typescript
       import { NextRequest, NextResponse } from 'next/server';
       import { existsSync } from 'fs';
       import { getDb, ensureSortedCrimesTable, getDbPath, isMockDataEnabled } from '@/lib/db';
       import type { Baseline168 } from '@/lib/signal-sources/contract';

       export const runtime = 'nodejs';
       export const dynamic = 'force-dynamic';

       const baselineCache = new Map<string, Baseline168>();

       async function buildBaselineFromDuckDB(duckdbPath: string): Promise<Baseline168> {
         const cached = baselineCache.get(duckdbPath);
         if (cached) return cached;

         if (isMockDataEnabled() || !existsSync(duckdbPath)) {
           throw new Error(`DuckDB cache not available at ${duckdbPath}; static baseline is the fallback.`);
         }

         const tableName = await ensureSortedCrimesTable();
         const db = await getDb();

         // Two queries: total_weeks + per-cell counts
         const totalWeeksRows = await new Promise<Array<{ total_weeks: number | null; n_events: number; ts_min: number | null; ts_max: number | null }>>((resolve, reject) => {
           db.all(
             `SELECT (EXTRACT(EPOCH FROM (MAX("Date") - MIN("Date"))) / (7 * 86400)) AS total_weeks,
                     COUNT(*) AS n_events,
                     MIN(EXTRACT(EPOCH FROM "Date")) AS ts_min,
                     MAX(EXTRACT(EPOCH FROM "Date")) AS ts_max
              FROM ${tableName}
              WHERE "Date" IS NOT NULL AND "Latitude" IS NOT NULL AND "Longitude" IS NOT NULL`,
             (err, rows) => err ? reject(err) : resolve(rows as Array<{ total_weeks: number | null; n_events: number; ts_min: number | null; ts_max: number | null }>)
           );
         });
         const totalWeeks = Number(totalWeeksRows[0]?.total_weeks ?? 0);
         const nEvents = Number(totalWeeksRows[0]?.n_events ?? 0);
         const tsMin = Number(totalWeeksRows[0]?.ts_min ?? 0);
         const tsMax = Number(totalWeeksRows[0]?.ts_max ?? 0);

         const cellRows = await new Promise<Array<{ hour: number; dow: number; count: number; mean_per_sec: number; sigma_per_sec: number }>>((resolve, reject) => {
           db.all(
             `SELECT EXTRACT(HOUR FROM "Date") AS hour,
                     EXTRACT(DOW FROM "Date") AS dow,
                     COUNT(*) AS count,
                     COUNT(*)::DOUBLE / (3600.0 * ?) AS mean_per_sec,
                     SQRT(COUNT(*)::DOUBLE / (3600.0 * ?)) / SQRT(3600.0 * ?) AS sigma_per_sec
              FROM ${tableName}
              WHERE "Date" IS NOT NULL AND "Latitude" IS NOT NULL AND "Longitude" IS NOT NULL
              GROUP BY hour, dow
              ORDER BY hour, dow`,
             (err, rows) => err ? reject(err) : resolve(rows as Array<{ hour: number; dow: number; count: number; mean_per_sec: number; sigma_per_sec: number }>),
             totalWeeks, totalWeeks, totalWeeks
           );
         });

         const cells = cellRows.map((row) => ({
           h: Number(row.hour),
           d: Number(row.dow),
           c: Number(row.count),
           mu: Number(row.mean_per_sec),
           sig: Number(row.sigma_per_sec),
         }));

         const baseline: Baseline168 = {
           header: {
             nEvents,
             tsMin,
             tsMax,
             totalWeeks,
             fingerprint: `sha256:duckdb-${tsMin}-${tsMax}`,
             builtAt: new Date().toISOString(),
           },
           cells,
         };

         baselineCache.set(duckdbPath, baseline);
         return baseline;
       }

       export async function GET(request: NextRequest) {
         try {
           const duckdbPath = getDbPath();
           const baseline = await buildBaselineFromDuckDB(duckdbPath);
           return NextResponse.json(baseline, {
             status: 200,
             headers: {
               'Cache-Control': 'public, s-maxage=3600, stale-while-revalidate=86400',
             },
           });
         } catch (error) {
           console.error('Contextual baseline route: DuckDB unavailable', error);
           return NextResponse.json(
             {
               error: 'baseline unavailable',
               source: 'duckdb',
               staticFile: '/baselines/baseline_168.json',
             },
             { status: 404 }
           );
         }
       }
       ```

       Notes:
       - The response shape matches the static JSON file EXACTLY so the client doesn't care which path produced the bytes.
       - The `Cache-Control: public, s-maxage=3600, stale-while-revalidate=86400` matches the research recommendation (1 hour fresh, 1 day stale).
       - The module-level `Map<string, Baseline168>` is keyed by `DUCKDB_PATH` per the research; the cache survives across requests within a single Node process.
       - The fingerprint for the DuckDB-derived baseline is `sha256:duckdb-{tsMin}-{tsMax}` (a poor-man's fingerprint; not cryptographic but identifies the dataset range). Real fingerprint comes from the static JSON.

    2. **Create `src/lib/signal-sources/baseline-loader.ts`** — client-side loader.

       ```typescript
       import type { Baseline168, Baseline168Winsorized } from './contract';

       let baselineCache: Baseline168 | null = null;
       let baselineWinsorizedCache: Baseline168Winsorized | null = null;
       let loadingPromise: Promise<Baseline168> | null = null;

       const STATIC_URL = '/baselines/baseline_168.json';
       const API_URL = '/api/adaptive/contextual-baseline';

       export async function loadBaseline168(): Promise<Baseline168> {
         if (baselineCache) return baselineCache;
         if (loadingPromise) return loadingPromise;
         loadingPromise = (async () => {
           try {
             const response = await fetch(STATIC_URL, { cache: 'force-cache' });
             if (!response.ok) {
               throw new Error(`Static baseline fetch failed: ${response.status}`);
             }
             const baseline = (await response.json()) as Baseline168;
             baselineCache = baseline;
             return baseline;
           } catch (staticError) {
             // Fall back to the API route (which may itself 404 if DuckDB is missing)
             const response = await fetch(API_URL);
             if (!response.ok) {
               throw new Error(`Both static and API baseline fetches failed; static: ${String(staticError)}; API: ${response.status}`);
             }
             const baseline = (await response.json()) as Baseline168;
             baselineCache = baseline;
             return baseline;
           }
         })();
         try {
           return await loadingPromise;
         } finally {
           loadingPromise = null;
         }
       }

       export function getBaseline168Sync(): Baseline168 | null {
         return baselineCache;
       }

       // Stub for 84-03: returns null until the winsorized baseline is computed.
       export function getBaseline168WinsorizedSync(): Baseline168Winsorized | null {
         return baselineWinsorizedCache;
       }
       ```

       Notes:
       - The loader tries the static file first (faster, no server round-trip), then the API route (DuckDB-derived).
       - The module-level cache (`baselineCache`) is the single source of truth for the sync accessor. The sync accessor returns `null` if the baseline hasn't been loaded yet — the dispatch returns `1.0` in that case.
       - The `loadingPromise` deduplicates concurrent calls (if two components call `loadBaseline168()` at the same time, they share the same promise).

       import { clampComparableWarpWeight } from '@/lib/binning/warp-scaling';
       import type { Baseline168 } from './contract';

       const clamp01 = (value: number): number => Math.min(1, Math.max(0, value));

       /**
        * Density mapper — Observed/Expected count ratio vs the 168-cell baseline.
        *
        * Formula: `clampComparableWarpWeight(1 + clamp01((O - E) / Math.max(E, 1)) * 1.5, 0.25, 4)`.
        * Range: [1.0, 2.5]. Returns 1.0 when:
        *   - The baseline is null (no data loaded yet — the slice creation falls back to neutral)
        *   - The cell is missing (defensive — should not happen with a complete 168-cell baseline)
        *   - O is approximately E (no deviation, clamp01 maps to 0)
        *
        * The formula maps observed/expected deviation to a [0, 1] linear range then to a [1, 2.5]
        * warpWeight. This means a bin matching the baseline gets 1.0 (neutral); a bin at 2x or
        * more gets the max 2.5. Sub-baseline bins (O < E) also get 1.0 — we don't compress
        * non-burst partitions; we only expand above-baseline bins.
        */
       export function densityWarpWeight(
         bin: { count: number },
         baseline: Baseline168 | null,
         h: number,
         d: number
       ): number {
         if (!baseline) return 1;
         const cell = baseline.cells[h * 7 + d];
         if (!cell) return 1;
         // E = mu * cell_seconds = mu * 3600 * total_weeks (the cell spans one hour of every week)
         const cellSeconds = 3600 * baseline.header.totalWeeks;
         const E = cell.mu * cellSeconds;
         if (!Number.isFinite(E) || E <= 0) return 1;
         const O = bin.count;
         const deviation = (O - E) / Math.max(E, 1);
         const positive = clamp01(deviation);
         return clampComparableWarpWeight(1 + positive * 1.5, 0.25, 4);
       }
