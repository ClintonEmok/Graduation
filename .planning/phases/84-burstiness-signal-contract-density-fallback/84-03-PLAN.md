    2. **Create `src/lib/signal-sources/winsorize.ts`** — the TS port. Per the research Q3 (L82-100) the recommendation is `d3-array.quantile` (R-7, linear interp, matches `numpy.percentile(method='linear')`).

       ```typescript
       import { quantile } from 'd3-array';

       /**
        * Winsorize an array of numbers at the [lowerPct, upperPct] quantiles.
        * Values below the lower quantile are clamped up; values above the upper quantile
        * are clamped down. Uses d3-array's `quantile` which implements the R-7 (linear)
        * interpolation — mathematically equivalent to `numpy.percentile(method='linear')`
        * for N >= 5. The 168-cell baseline has 168 floats so the cross-language match
        * is exact within 1e-6 (verified by the test in `winsorize.test.ts`).
        */
       export function winsorize(values: number[], lowerPct: number, upperPct: number): number[] {
         if (values.length === 0) return [];
         const lo = quantile(values, lowerPct) ?? -Infinity;
         const hi = quantile(values, upperPct) ?? Infinity;
         return values.map((v) => Math.min(hi, Math.max(lo, v)));
       }
       ```

       ```typescript
         const sigsW = winsorize(sigs, 0.05, 0.95);
         return {
           cells: baseline.cells.map((c, i) => ({
             h: c.h,
             d: c.d,
             muW: meansW[i] ?? c.mu,
             sigW: sigsW[i] ?? c.sig,
           })),
         };
       }

       /**
        * Contextual mapper — winsorized Pearson residual against the 168-cell baseline.
        *
        * z = (count - muW * windowSec) / max(sigW * sqrt(windowSec), EPSILON)
        * z_clamped = clamp(z, -2, 5)
        * warpWeight = WARP_MIN + (z_clamped - Z_CLAMP_MIN) / (Z_CLAMP_MAX - Z_CLAMP_MIN) * (WARP_MAX - WARP_MIN)
        *
        * z = 0 (observed matches baseline) → warpWeight = 0.2 + (0+2)/7 * 2.8 = 1.0
        * z = 5 (max burst) → warpWeight = 3.0
        * z = -2 (max compression) → warpWeight = 0.2
        *
        * Returns 1.0 when no baseline is loaded (stub fallback) or the cell is missing.
        */
       export function contextualWarpWeight(
         bin: { count: number; startTime: number; endTime: number },
         baselineWinsorized: Baseline168Winsorized | null,
         h: number,
         d: number
       ): number {
         if (!baselineWinsorized) return 1;
         const cell = baselineWinsorized.cells[h * 7 + d];
         if (!cell) return 1;
         const windowSec = (bin.endTime - bin.startTime) / 1000; // bin times are in ms
         if (!Number.isFinite(windowSec) || windowSec <= 0) return 1;
         const muW = cell.muW;
         const sigW = cell.sigW;
         const expected = muW * windowSec;
         const sigma = sigW * Math.sqrt(windowSec);
         const z = (bin.count - expected) / Math.max(sigma, EPSILON);
         const zClamped = Math.max(Z_CLAMP_MIN, Math.min(Z_CLAMP_MAX, z));
         const t = (zClamped - Z_CLAMP_MIN) / (Z_CLAMP_MAX - Z_CLAMP_MIN); // [0, 1]
         return WARP_MIN + t * (WARP_MAX - WARP_MIN);
       }
       ```

       Notes:
       - The `bin` param is typed structurally as `{ count: number; startTime: number; endTime: number }` — matches the `TimeBin` interface fields without importing the binning types module.
       - `windowSec` conversion: `bin.startTime` and `bin.endTime` are in epoch ms (per `src/lib/binning/types.ts:13-15`); divide by 1000 to get seconds.
       - The z formula `z = (count - muW * windowSec) / max(sigW * sqrt(windowSec), EPSILON)` matches `metrics/contextual.py:180` for a single-cell window: `z = (n_events - mu_per_window) / np.sqrt(mu_safe)`. For a single bin spanning one cell, `mu_per_window = muW * 3600` (for 1h) and `sigma = sqrt(mu_per_window)`. For multi-cell bins, the per-window sum generalises — but since `bin.startTime` and `bin.endTime` give us the exact window, we can compute directly without per-cell summation.
       - The z-clamp range `[-2, 5]` and warpWeight range `[0.2, 3.0]` are the locked values from CONTEXT L19. The linear remap at `z = 0` gives exactly `1.0` (0.2 + 2/7 * 2.8 = 0.2 + 0.8 = 1.0), preserving the property that a bin matching the baseline gets the neutral 1.0 weight.
       - The `EPSILON` guard prevents divide-by-zero on degenerate sigma cells (matches `metrics/contextual.py:51 EPSILON = 1e-9`).
