    import type { Suggestion } from '@/types/suggestion';
    
    export type SuggestionEventType = 'accept' | 'reject' | 'modify' | 'add' | 'clear';
    
    export interface SuggestionEvent {
      type: SuggestionEventType;
      suggestion?: Suggestion;
      timestamp: number;
    }
    
    type EventHandler = (event: SuggestionEvent) => void;
    
    const handlers = new Set<EventHandler>();
    
    export const suggestionEvents = {
      subscribe(handler: EventHandler): () => void {
        handlers.add(handler);
        return () => handlers.delete(handler);
      },
      
      dispatch(event: SuggestionEvent): void {
        handlers.forEach((handler) => handler(event));
      },
      
      emitAccept(suggestion: Suggestion): void {
        this.dispatch({ type: 'accept', suggestion, timestamp: Date.now() });
      },
      
      emitReject(suggestion: Suggestion): void {
        this.dispatch({ type: 'reject', suggestion, timestamp: Date.now() });
      },
      
      emitModify(suggestion: Suggestion): void {
        this.dispatch({ type: 'modify', suggestion, timestamp: Date.now() });
      },
      
      emitAdd(suggestion: Suggestion): void {
        this.dispatch({ type: 'add', suggestion, timestamp: Date.now() });
      },
      
      emitClear(): void {
        this.dispatch({ type: 'clear', timestamp: Date.now() });
      },
    };
    ```

    };
    
    interface HistoryStore {
      acceptedHistory: HistoryEntry[];
      lastAction: UndoAction | null;
      undoTimeout: ReturnType<typeof setTimeout> | null;
      showUndoToast: boolean;
      
      addToHistory: (suggestion: Suggestion) => void;
      clearHistory: () => void;
      reapplyFromHistory: (historyId: string) => HistoryEntry | null;
      setUndoToast: (show: boolean) => void;
      setUndoTimeout: (timeout: ReturnType<typeof setTimeout> | null) => void;
    }
    
    export const useSuggestionHistoryStore = create<HistoryStore>((set, get) => ({
      acceptedHistory: [],
      lastAction: null,
      undoTimeout: null,
      showUndoToast: false,
      
      addToHistory: (suggestion) =>
        set((state) => ({
          acceptedHistory: [
            {
              id: crypto.randomUUID(),
              suggestion,
              acceptedAt: Date.now(),
              contextMetadata: suggestion.contextMetadata,
            },
            ...state.acceptedHistory,
          ].slice(0, 50), // Keep last 50 entries
        })),
      
      clearHistory: () => set({ acceptedHistory: [] }),
      
      reapplyFromHistory: (historyId) => {
        const entry = get().acceptedHistory.find((h) => h.id === historyId);
        return entry || null;
      },
      
      setUndoToast: (show) => set({ showUndoToast: show }),
      
      setUndoTimeout: (timeout) => set({ undoTimeout: timeout }),
    }));

    interface ComparisonStore {
      comparisonIds: [string | null, string | null];
      
      setComparisonId: (index: 0 | 1, id: string | null) => void;
      clearComparison: () => void;
      canAddComparison: () => boolean;
      isCompared: (id: string) => boolean;
    }
    
    export const useSuggestionComparisonStore = create<ComparisonStore>((set, get) => ({
      comparisonIds: [null, null],
      
      setComparisonId: (index, id) =>
        set((state) => {
          const newIds: [string | null, string | null] = [...state.comparisonIds];
          newIds[index] = id;
          return { comparisonIds: newIds };
        }),
      
      clearComparison: () => set({ comparisonIds: [null, null] }),
      
      canAddComparison: () => {
        const [id1, id2] = get().comparisonIds;
        return id1 === null || id2 === null;
      },
      
      isCompared: (id) => {
        const [id1, id2] = get().comparisonIds;
        return id1 === id || id2 === id;
      },
    }));
    ```
